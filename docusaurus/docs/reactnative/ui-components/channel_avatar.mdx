The `ChannelAvatar` is the default UI component for rendering avatar/image for channel within `ChannelList` component.

You can set `PreviewAvatar` prop on `ChannelList` component to replace this default component:

```jsx
import { 
  ChannelPreviewMessenger,
  useChannelPreviewDisplayAvatar,
  useChannelPreviewDisplayPresence,
} from 'stream-chat-react-native';

import { CustomGroupAvatar } from '...'; // your custom group avatar component
import { CustomAvatar } from '...'; // your custom avatar component

const CustomPreviewAvatar = ({ channel }) => {
  const displayAvatar = useChannelPreviewDisplayAvatar(channel);
  const displayPresence = useChannelPreviewDisplayPresence(channel);

  if (displayAvatar.images) {
    return <CustomGroupAvatar images={displayAvatar.images} names={displayAvatar.names} size={40} />;
  }

  return (
    <CustomAvatar
      image={displayAvatar.image}
      name={displayAvatar.name}
      online={displayPresence}
      size={40}
    />
  );
}

<ChannelList PreviewAvatar={CustomPreviewAvatar} />
```
